━━━━━━━━━━━━━━━━━━━━━━━━
guidance_config = {
    'name': 'global_target_pos',
    'params': {
        'target_pos': [[x1, y1], [x2, y2], ...],  # Waypoints
        'urgency': [u1, u2, ...],                 # 0-1 for each agent
        'pref_speed': 1.42,                       # m/s
        'dt': 0.1,                                # seconds
        'min_progress_dist': 0.5,                 # meters
    },
    'agents': [0, 1, 2, ...]  # Agent indices
}
guidance_config_list = [guidance_config]  # Wrap in list


VALID GUIDANCE TYPES:
━━━━━━━━━━━━━━━━━━━━━━
✓ 'global_target_pos'         → Recommended for waypoints
✓ 'global_target_pos_at_time' → With time constraint
✓ 'target_pos'                → Local frame variant
✓ 'agent_collision'           → Avoid other agents
✓ 'map_collision'             → Stay on valid areas
✓ 'target_speed'              → Maintain speed
✓ 'social_group'              → Keep agents together
